#include <iostream>
using namespace std;

class BankAccount {
private:
    double balance;  

public:
    void setBalance(double b) {
        if (b >= 0)
            balance = b;
        else
            cout << "Invalid balance amount!" << endl;
    }

    double getBalance() {
        return balance;
    }
};

int main() {
    BankAccount account;

    account.setBalance(69420.50);

    cout << "Account Balance: " << account.getBalance() << endl;
    system("pause");    
    return 0;
}
