#include <iostream>
using namespace std;

class Teacher {
public:
    void teach() {
        cout << "Teaching students " << endl;
    }
};

class Researcher {
public:
    void research() {
        cout << "Doing research work " << endl;
    }
};

class Professor : public Teacher, public Researcher {
public:
    void guide() {
        cout << "Guiding students and research work " << endl;
    }
};

int main() {
    Professor prof;

    prof.teach();      
    prof.research();   
    prof.guide();   

    system ("pause");
    return 0;
}
