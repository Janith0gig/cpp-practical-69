#include <iostream>
using namespace std;

class Math {
public:
    int add(int a, int b) {
        return a + b;
    }

    int add(int a, int b, int c) {
        return a + b + c;
    }

    double add(double a, double b) {
        return a + b;
    }
};

int main() {
    Math m;

    // Calling different versions of add()
    cout << "Sum of two integers: " << m.add(40, 27) << endl;
    cout << "Sum of three integers: " << m.add(22, 9, 7) << endl;
    cout << "Sum of two doubles: " << m.add(4.5, 2.4) << endl;
    system ("pause");
    return 0;
}
